лимиты.js
// 🎯 Автораспределение любого поступления
// Срабатывает при каждом входящем переводе

vault.onReceive(async (tx) => {
  const { amount, from } = tx;

  // Распределяем по карманам
  await vault.send({
    to: 'savings',
    amount: amount * 0.30,
    note: 'Накопления 30%'
  });

  await vault.send({
    to: 'tax_reserve',
    amount: amount * 0.20,
    note: 'Налоговый резерв'
  });

  // Уведомление в Telegram
  await vault.notify({
    message: `💰 Получено ${amount} USDT от ${from}`
  });
});
// 📅 Автоматические подписки
// Списывает деньги по расписанию

vault.schedule({
  name: 'Netflix',
  cron: '0 9 15 * *', // 15-го каждый месяц
  action: async () => {
    // Проверяем баланс перед списанием
    if (vault.balance < 50) {
      await vault.notify({
        message: '⚠️ Мало средств для Netflix'
      });
      return;
    }

    await vault.send({
      to: 'UQC...netflix',
      amount: 15,
      currency: 'USDT'
    });
  }
});
// 🚨 Лимиты и защита от трат
// Блокирует крупные переводы

vault.onTransfer(async (tx, next) => {
  const dailySpent = await vault.getDailySpent();

  // Дневной лимит 500 USDT
  if (dailySpent + tx.amount > 500) {
    throw new Error('Превышен дневной лимит');
  }

  // Для сумм >1000 нужно 2 подтверждения
  if (tx.amount > 1000) {
    await vault.requireMultiSig({
      signers: ['trusted_contact_1'],
      threshold: 1
    });
  }

  next(); // Пропускаем транзакцию
});
onReceive

Хук на получение денег

Срабатывает автоматически при каждом входящем переводе. Можно распределять деньги, отправлять уведомления, логировать.

schedule

Расписание платежей

Cron-синтаксис для настройки любых повторяющихся платежей. Подписки, аренда, зарплаты — всё автоматически.

onTransfer

Фильтр исходящих

Перехватывает каждый перевод перед отправкой. Можно добавить лимиты, мультиподпись, проверку адреса.

Vault

Изолированные карманы

Каждый карман — отдельный смарт-контракт со своими правилами. Деньги физически разделены на блокчейне.

Как это работает

Четыре шага до
твоего банка

01

Создаёшь кошелёк

Генерируется приватный ключ — он хранится только у тебя. Мы его никогда не видим.

02

Деплоишь контракт

Твой личный смарт-контракт записывается в TON блокчейн. Теперь это математически твоё.

03

Настраиваешь правила

Через UI или SDK пишешь правила: как распределять деньги, когда списывать, кому доверять.

04

Получаешь долю

Каждую неделю получаешь часть комиссий всей сети пропорционально твоей активности.
